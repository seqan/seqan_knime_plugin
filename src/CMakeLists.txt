# We need plugin generation
include (generate_plugin)

set (UPDATE_SITE_VERSION_MAJOR "1")
set (UPDATE_SITE_VERSION_MINOR "0")
set (UPDATE_SITE_VERSION_PATCH "0")

add_update_site("SeqAn" "de.seqan.update")
